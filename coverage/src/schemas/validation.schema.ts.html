import { z } from "zod/v4";
 
/**
 * When validation should be triggered
 */
export const ValidationTiming = z.enum([
  "onChange",
  "onBlur",
  "onSubmit",
  "manual",
]);
 
/**
 * Validation strategies for form-level coordination
 */
export const ValidationStrategy = z.enum([
  "eager", // Show errors immediately
  "lazy", // Only show errors after first submission attempt
  "progressive", // Show errors as user progresses through form
]);
 
/**
 * Severity levels for validation messages
 */
export const ValidationSeverity = z.enum([
  "error", // Blocks form submission
  "warning", // Doesn't block but shows warning
]);
 
/**
 * Cross-field validation rule
 */
export const CrossFieldValidationRule = z.object({
  id: z.string(),
  name: z.string().optional(),
 
  // Fields this validation applies to
  fields: z.array(z.string()),
 
  // Validation logic
  validator: z.function(), // (fieldValues) => { isValid: boolean, message?: string }
 
  // When to run this validation
  timing: ValidationTiming,
 
  // Error details
  severity: ValidationSeverity,
  message: z.string(),
});
 
/**
 * Form-level validation configuration
 */
export const FormValidationConfig = z.object({
  strategy: ValidationStrategy,
 
  // Cross-field validation rules
  crossFieldRules: z.array(CrossFieldValidationRule).optional(),
 
  // Should we validate all fields before allowing submission?
  validateAllOnSubmit: z.boolean().default(true),
 
  // Should we stop validation on first error?
  stopOnFirstError: z.boolean().default(false),
});
 
/**
 * Validation state for a single field
 */
export const FieldValidationState = z.object({
  isValid: z.boolean(),
  isDirty: z.boolean(),
  isTouched: z.boolean(),
  errors: z.array(z.string()),
  warnings: z.array(z.string()),
});
 
/**
 * Overall form validation state
 */
export const FormValidationState = z.object({
  isValid: z.boolean(),
  isSubmitting: z.boolean(),
  hasSubmitted: z.boolean(),
 
  // Field states
  fields: z.record(z.string(), FieldValidationState),
 
  // Form-level errors from cross-field validation
  formErrors: z.array(z.string()),
});
 
export type ValidationTiming = z.infer<typeof ValidationTiming>;
export type ValidationStrategy = z.infer<typeof ValidationStrategy>;
export type ValidationSeverity = z.infer<typeof ValidationSeverity>;
export type CrossFieldValidationRule = z.infer<typeof CrossFieldValidationRule>;
export type FormValidationConfig = z.infer<typeof FormValidationConfig>;
export type FieldValidationState = z.infer<typeof FieldValidationState>;
export type FormValidationState = z.infer<typeof FormValidationState>;
